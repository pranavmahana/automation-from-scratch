/** * Retry-safe click. Waits for visible + enabled before clicking. * Handles flaky elements in CI environments. */ async safeClick(selector, options = {}) { const el = this.page.locator(selector).first(); await el.waitFor({ state: 'visible', timeout: this.timeout }); await el.scrollIntoViewIfNeeded(); await el.click(options); } // Mock any API — test UI in isolation, no backend needed async mockAPI(url, body, status = 200) { await this.page.route(url, route => route.fulfill({ status, body: JSON.stringify(body) }) ); }
